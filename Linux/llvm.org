缩小成35G:
resize2fs /dev/vg1/java 34G
lvreduce /dev/vg1/java -L 35G
resize2fs /dev/vg1/java

扩大成35G
lvextend /dev/vg1/java -L 35G
resize2fs /dev/vg1/java

加大1G
lvextend /dev/vg1/java -L +1G
resize2fs /dev/vg1/java
