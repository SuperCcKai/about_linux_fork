  /* 位的与或非，是针对int 类型的。对于char short 会被自动提升为int 之后再做与或非 */
  /* ~ 取返 */

#+BEGIN_SRC c
#include <stdio.h>
#include <stdlib.h>
#include <malloc.h>
#include <string.h>
int main(int argc, char *argv[]){

  unsigned int a, b, mask = 0x0000ff00;
  a = 0x12345678;               /* 针对a的对8～15位进行操作， */
  b = (a & mask) >> 8; /* 0x00000056 */
  printf ("%x\n",b);

  b =( a>>8) &(~((~0 )<<8 ));   /* ~0== ffffffff , (~0)<<8 == ffffff00 */
  printf ("%x\n",b);

  return 0;
}
